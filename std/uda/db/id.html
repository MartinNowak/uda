    Struct id

    Marks a field as primary key.

    struct id ;

    Example

    struct User
{
    @id string email;
}

struct Entry
{
    int id;
}

struct Person
{
    import std.datetime : Date;

    // compound key
    static struct Key
    {
        string firstname, surname;
        Date birthday;
    }

    @id Key key;
}
